# Event organizer for the centaur tolerance events - should be a huge improvement to performance - having 1 overall check for having the minority instead of one for EACH new event that gets added
country_event = {
	id = centaur_tolerance_events.0
	title = race_setup_events.1.d
	desc = race_setup_events.1.d
	picture = BORDER_TENSION_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		fast_owns_num_pops_of_race = { race = centaur amount = 1 }
		NOT = { has_country_modifier = centaur_administration }
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 10
			has_country_modifier = racial_focus_centaur
		}
		modifier = {
			factor = 1.5
			has_government_attribute = more_migration_and_tolerance
		}
	}

	option = {
		random_list = {
			1 = { }
			# ensure there's an empty option first in case none of the others are available
			# and also because it seems this one might be picked rarely even if the others are available...

			#generic demand autonomy
			10 = {
				trigger = {
					any_owned_province = {
						NOT = { local_autonomy = 50 }
						is_capital = no
						has_centaur_minority_trigger = yes
					}
				}
				country_event = { id = centaur_tolerance_events.1 }
			}
			
			# generic tolerance increase
			10 = {
				trigger = {
					OR = {
						stability = 1
						in_golden_age = yes
						ruler_has_personality = tolerant_personality
						ruler_has_personality = kind_hearted_personality
						ruler_has_personality = benevolent_personality
						has_idea_group = humanist_ideas
						
						any_ally = {
							has_country_modifier = centaur_administration
						}
					}
				}
				country_event = { id = centaur_tolerance_events.2 }
			}
			# generic tolerance decrease
			10 = {
				trigger = {
					OR = {
						NOT = { stability = 0 }
						is_bankrupt = yes
						inflation = 25
						AND = {
							is_at_war = yes
							NOT = { war_score = 20 }
						}
						war_exhaustion = 10
						ruler_has_personality = cruel_personality
						ruler_has_personality = malevolent_personality
						ruler_has_personality = conqueror_personality
						
						any_rival_country = {
							has_country_modifier = centaur_administration
						}
					}
				}
				country_event = { id = centaur_tolerance_events.3 }
			}

			# centaur minister
			10 = {
				modifier = {
					factor = 1.5
					medium_tolerance_centaur_race_trigger = yes
				}
				modifier = {
					factor = 2
					high_tolerance_centaur_race_trigger = yes
				}
				country_event = {
					id = centaur_tolerance_events.4
				}
			}

			# migrating centaurs trample fields
			10 = {
				modifier = {
					factor = 0.5
					medium_tolerance_centaur_race_trigger = yes
				}
				if = { # prefers fields if possible
					limit = {
						any_owned_province = {
							OR = {
								has_terrain = farmlands
								has_terrain = steppe
								has_terrain = grasslands
							}
							fast_has_pop_of_race = { race = centaur }
						}
					}
					random_owned_province = {
						limit = {
							OR = {
								has_terrain = farmlands
								has_terrain = steppe
								has_terrain = grasslands
							}
							fast_has_pop_of_race = { race = centaur }
						}
						province_event = { id = centaur_tolerance_events.5 }
					}
				}
				else = {
					random_owned_province = {
						limit = { fast_has_pop_of_race = { race = centaur } }
						province_event = { id = centaur_tolerance_events.5 }
					}
				}
			}

			# centaur haulers outperform beasts of burden
			10 = {
				trigger = { 
					any_owned_province = { # Has an "agricultural" trade good AND Centaurs present
						fast_has_pop_of_race = { race = centaur }
						OR = {
							trade_goods = grain
							trade_goods = fish
							trade_goods = livestock
							trade_goods = wool
						}	
					}
				}
				modifier = {
					factor = 2
					medium_tolerance_centaur_race_trigger = yes
				}
				random_owned_province = {
					limit = {
						fast_has_pop_of_race = { race = centaur }
						OR = {
							trade_goods = grain
							trade_goods = fish
							trade_goods = livestock
							trade_goods = wool
						}
					}
					province_event = { id = centaur_tolerance_events.6 }
				}
			}

			#same day shipping
			10 = {
				trigger = {
					has_estate = estate_burghers
				}
				modifier = {
					factor = 1.5
					medium_tolerance_centaur_race_trigger = yes
				}
				modifier = {
					factor = 2
					high_tolerance_centaur_race_trigger = yes
				}
				country_event = { id = centaur_tolerance_events.7 }
			}

			#urbanizing centaurs
			10 = {
				trigger = {
					any_owned_province = {
						development = 15
						has_centaur_minority_trigger = yes
					}
				}
				modifier = {
					factor = 1.5
					medium_tolerance_centaur_race_trigger = yes
				}
				modifier = {
					factor = 2
					high_tolerance_centaur_race_trigger = yes
				}
				random_owned_province = {
					limit = {
						development = 15
						has_centaur_minority_trigger = yes
					}
					province_event = { id = centaur_tolerance_events.8 }
				}	
			}

			#bandit and steed ravage countryside
			10 = {
				modifier = {
					factor = 2
					low_tolerance_centaur_race_trigger = yes
				}
				modifier = { # more banditry if unstable 
					factor = 2
					NOT = { stability = 0 }
				}
				modifier = {
					factor = 0.5
					high_tolerance_centaur_race_trigger = yes
				}
				modifier = { # less banditry if stable 
					factor = 0.5
					stability = 2
				}

				random_owned_province = {
					limit = { fast_has_pop_of_race = { race = centaur } }
					province_event = { id = centaur_tolerance_events.9 }
				}
				
			}

			#centaur festival disrupts peace and tranquility
			10 = {
				country_event = { id = centaur_tolerance_events.10 }
			}

			#centaur traders approach markets
			10 = {
				modifier = {
					factor = 1.5
					medium_tolerance_centaur_race_trigger = yes
				}
				modifier = { # less likely if very little trade
					factor = 0.5
					NOT = { trade_income_percentage = 0.05 }
				}
				modifier = { # more likely if lots of trade
					factor = 2
					trade_income_percentage = 0.30
				}

				country_event = { id = centaur_tolerance_events.11 }
			}

			#noble attempts to ride centaur
			10 = {
				trigger = { 
					has_estate = estate_nobles
				}
				modifier = {
					factor = 0.5
					high_tolerance_centaur_race_trigger = yes # less likely to occur if your people are already well integrated with Centaurs
				}
				country_event = { id = centaur_tolerance_events.12 }
			}

			30 = {
				trigger = {
					ai = no
					OR = {
						has_country_modifier = just_lost_war_timer
						average_unrest = 10
						is_bankrupt = yes
						NOT = { stability = 0 }
					}
					NOT = { high_tolerance_centaur_race_trigger = yes }
					is_not_purging_or_expelling_any_race_trigger = yes
				}
				set_country_flag = propflag_centaur
				country_event = { id = tolerance_generic.1 }
			}
		}
	}
}
