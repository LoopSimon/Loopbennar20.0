namespace = charkuchin_kobold

country_event = {
	id = charkuchin_kobold.0
	title = charkuchin_kobold.0.t
	desc = charkuchin_kobold.0.d
	picture = TRADEGOODS_eventPicture

	is_triggered_only = yes

	trigger = {
		OR = {
			culture = yawasichya_kobold
			culture = jatukyu_kobold
			culture = unitsan_kobold
		}
	}

	option = {
		name = charkuchin_kobold.0.a
		ai_chance = { factor = 1 }
		if = {
			limit = {
				ai = yes
			}
			add_country_modifier = {
				name = charkuchin_expansion_ai
				duration = -1
			}
			add_country_modifier = {
				name = charkuchin_expansion_ai_help
				duration = 72
			}
		}
		else = {
			add_country_modifier = {
				name = charkuchin_expansion
				duration = -1
			}
		}
	}
}
