mechanim_advancement_scripted_effect = {
	change_variable = {
		which = mechanim_progress_level
		value = 1
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 9
			}
		}
		set_country_flag = mechanim_doubled_dormancy
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 8
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 9
				}
			}
		}
		set_country_flag = mechanim_free_permanent_damestear
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 7
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 8
				}
			}
		}
		mission_reward_artifice_points_10 = yes
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 6
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 7
				}
			}
		}
		set_country_flag = mechanim_free_feudalism
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 5
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 6
				}
			}
		}
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 4
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 5
				}
			}
		}
		add_treasury = 150
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 3
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 4
				}
			}
		}
		add_reform_progress_medium_effect = yes
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 2
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 3
				}
			}
		}
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_progress_level
				value = 1
			}
			NOT = {
				check_variable = {
					which = mechanim_progress_level
					value = 2
				}
			}
		}
		add_treasury = 100
	}
}
