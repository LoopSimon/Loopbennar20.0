the_fightback = {
	sprite = "GFX_sky_domain_denounce_yrilak"
	cost = 200
	
	trigger = {
		ROOT = {
			is_native_tribe = no
		}
	}

	potential = {
	}

	effect = {
		custom_tooltip = the_fightback_tt
		hidden_effect = {
			set_variable = {
				which = ForLoopCounter
				value = 100
			}
			every_owned_province = {
				limit = {
					fast_has_pop_of_race = { race = mechanim }
					fast_has_pop_of_race = { race = human }
					PREV = { check_variable = { ForLoopCounter = 1 } }
					NOT = { has_province_modifier = dunborgula_the_fightback }
				}
				add_province_modifier = {
					name = dunborgula_the_fightback
					duration = 1825
					desc = dunborgula_the_fightback_desc
				}
				PREV = {
					subtract_variable = {
						which = ForLoopCounter
						value = 1
					}
				}
			}
		}
	}
}
