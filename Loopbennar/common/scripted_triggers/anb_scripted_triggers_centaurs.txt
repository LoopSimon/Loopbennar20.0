#Possible Types: 'Tournament', 'Festival', 'Rite', 'Communion', 'ReligiousAction'
irdaeos_religious_actions_completed = {
	custom_trigger_tooltip = {
		tooltip = $type$s_centaur_$amount$_tooltip
		check_variable = {
			which = nb$type$s
			value = $amount$
		}
	}
}

has_base_nomadic_zone_of_control_modifier = {
	custom_trigger_tooltip = {
		tooltip = have_base_nomadic_zone_of_control_tt
		OR = {
			has_province_modifier = base_nomadic_zone_of_control
			has_province_modifier = base_nomadic_zone_of_control_double
		}
	}
}
